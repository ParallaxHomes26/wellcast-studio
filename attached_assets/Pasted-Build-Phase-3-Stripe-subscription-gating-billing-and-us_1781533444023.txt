5. Create a subscription check utility (src/lib/subscription.ts)
typescriptexport type SubscriptionTier = 'trialing' | 'basic' | 'starter' | 'pro' | 'founding_member' | 'expired' | 'canceled'

export function getSubscriptionTier(profile: any): SubscriptionTier {
  if (!profile) return 'expired'
  
  // Check if trial is active
  if (profile.subscription_status === 'trialing') {
    const trialEnd = new Date(profile.trial_ends_at)
    if (trialEnd > new Date()) return 'trialing'
    else return 'expired'
  }
  
  // Check founding member
  if (profile.founding_member && profile.subscription_status === 'active') {
    return 'founding_member'
  }
  
  // Check active subscription by price ID
  if (profile.subscription_status === 'active') {
    const priceId = profile.subscription_price_id
    if ([
      'price_1TibHO2Zpe8a4y2MgNN4wVrI',
      'price_1TibL22Zpe8a4y2MWJx4WYFY'
    ].includes(priceId)) return 'basic'
    
    if ([
      'price_1TibHw2Zpe8a4y2MouaBgTQW',
      'price_1TibLg2Zpe8a4y2M2MWuaYjV'
    ].includes(priceId)) return 'starter'
    
    if ([
      'price_1TibIO2Zpe8a4y2MW1yIg27S',
      'price_1TibM02Zpe8a4y2MiKl4WaI5'
    ].includes(priceId)) return 'pro'
    
    if (priceId === 'price_1TibKD2Zpe8a4y2MLI3ii0KD') return 'founding_member'
  }
  
  return 'expired'
}

export function getRunLimit(tier: SubscriptionTier): number | 'unlimited' {
  switch (tier) {
    case 'trialing': return 3 // 3 runs during trial
    case 'basic': return 2
    case 'starter': return 4
    case 'pro': return 'unlimited'
    case 'founding_member': return 'unlimited'
    default: return 0
  }
}

export function canRunGeneration(profile: any): { allowed: boolean, reason?: string } {
  const tier = getSubscriptionTier(profile)
  const limit = getRunLimit(tier)
  
  if (tier === 'expired' || tier === 'canceled') {
    return { allowed: false, reason: 'Your trial has ended. Please subscribe to continue.' }
  }
  
  if (limit === 'unlimited') return { allowed: true }
  
  if (profile.run_count_this_month >= limit) {
    return { 
      allowed: false, 
      reason: `You've used all ${limit} runs this month. Upgrade to get more.`
    }
  }
  
  return { allowed: true }
}

12. Add run count reset logic
In the generate API route, after a successful run:
typescript// Check if run_count_reset_at is in a previous month — if so reset the counter
const resetDate = new Date(profile.run_count_reset_at)
const now = new Date()
const needsReset = resetDate.getMonth() !== now.getMonth() || 
                   resetDate.getFullYear() !== now.getFullYear()

if (needsReset) {
  await supabaseAdmin
    .from('profiles')
    .update({ 
      run_count_this_month: 1,
      run_count_reset_at: now.toISOString()
    })
    .eq('id', userId)
} else {
  await supabaseAdmin
    .from('profiles')
    .update({ 
      run_count_this_month: profile.run_count_this_month + 1 
    })
    .eq('id', userId)
}
